108def pack_epub(directory, file): 

109 """Pack the given ``directory`` into an epub (i.e. zip) archive 

110 given as ``file``, which can be a file-path or file-like object. 

111 """ 

112 with zipfile.ZipFile(file, 'w', zipfile.ZIP_DEFLATED) as zippy: 

113 base_path = os.path.abspath(directory) 

114 for root, dirs, filenames in os.walk(directory): 

115 # Strip the absolute path 

116 archive_path = os.path.relpath(root, base_path) 

117 for filename in filenames: 

118 filepath = os.path.join(root, filename) 

119 archival_filepath = os.path.join(archive_path, filename) 

120 zippy.write(filepath, archival_filepath)

133class EPUB(Sequence): 

134 """Represents an EPUB3 file structure in object form. 

135 It is designed to work with .epub files (zip files), 

136 but will uncompress them to a temporary location before working with them. 

137 

138 """ 

139 

140 def __init__(self, packages=None, root=None): 

141 self._packages = packages is None and [] or packages 

142 self._root = root 

143 

144 @classmethod 

145 def from_file(cls, file): 

146 """Create the object from a *file* or *file-like object*. 

147 The file can point to an ``.epub`` file or a directory 

148 (the contents of which reflect 

149 the internal struture of an ``.epub`` archive). 

150 If given an non-archive file, 

151 this structure will be used when reading in and parsing the epub. 

152 If an archive file is given, 

153 it will be extracted to the temporal filesystem. 

154 """ 

155 root = None 

156 if zipfile.is_zipfile(file): 

157 unpack_dir = tempfile.mkdtemp('-epub') 

158 # Extract the epub to the current working directory. 

159 with zipfile.ZipFile(file, 'r') as zf: 

160 zf.extractall(path=unpack_dir) 

161 root = unpack_dir 

162 elif os.path.isdir(file): 

163 root = file 

164 else: 

165 raise TypeError("Can't decipher what should be done " 

166 "with the given file.") 

167 

168 # NOTE We ignore the mimetype file, as it's not extremely important 

169 # to anything done here. 

170 

171 # Build a blank epub object then parse the packages. 

172 container_xml_filepath = os.path.join(root, 

173 EPUB_CONTAINER_XML_RELATIVE_PATH) 

174 container_xml = etree.parse(container_xml_filepath) 

175 

176 packages = [] 

177 for pkg_filepath in container_xml.xpath( 

178 '//ns:rootfile/@full-path', 

179 namespaces=EPUB_CONTAINER_XML_NAMESPACES): 

180 filepath = os.path.join(root, pkg_filepath) 

181 packages.append(Package.from_file(filepath)) 

182 return cls(packages=packages, root=root) 

183 

184 @staticmethod 

185 def to_file(epub, file): 

186 """Export to ``file``, which is a *file* or *file-like object*.""" 

187 directory = tempfile.mkdtemp('-epub') 

188 # Write out the contents to the filesystem. 

189 package_filenames = [] 

190 for package in epub: 

191 opf_filepath = Package.to_file(package, directory) 

192 opf_filename = os.path.basename(opf_filepath) 

193 package_filenames.append(opf_filename) 

194 

195 # Create the container.xml 

196 container_xml_filepath = os.path.join(directory, 

197 EPUB_CONTAINER_XML_RELATIVE_PATH) 

198 template = jinja2.Template(CONTAINER_XML_TEMPLATE, 

199 trim_blocks=True, lstrip_blocks=True) 

200 os.makedirs(os.path.dirname(container_xml_filepath)) # FIXME PY3 

201 with open(container_xml_filepath, 'w') as fb: 

202 xml = template.render(package_filenames=package_filenames) 

203 fb.write(xml) 

204 # Write the mimetype file. 

205 with open(os.path.join(directory, 'mimetype'), 'w') as fb: 

206 fb.write("application/epub+zip") 

207 

208 # Pack everything up 

209 pack_epub(directory, file=file)

219class OPFParser: 

220 """Parse an ``.opf`` xml document's metadata. 

221 This class is callable to respond as a function 

222 after inititiation as a singleton. 

223 The work for parsing the metadata is methodized to provide 

224 detailed exceptions. 

225 

226 """ 

227 namespaces = EPUB_OPF_NAMESPACES 

228 metadata_required_keys = ( 

229 'publisher', 'publication_message', 

230 ) 

231 metadata_optional_keys = ( 

232 'title', 'identifier', 'language', 'license_text', 'license_url', 

233 ) 

234 

235 def __init__(self, opf_xml): 

236 """Given the opf_xml (an ``lxml.etree.ElementTree``), 

237 parse the metadata fields on access. 

238 """ 

239 self._xml = opf_xml 

240 

241 def parser(self, xpath, prefix="/opf:package/opf:metadata/"): 

242 metadata_prefix = "/opf:package/opf:metadata/" 

243 values = self._xml.xpath(metadata_prefix + xpath, 

244 namespaces=self.namespaces) 

245 return values 

246

253 value = None 

254 return value

310 @property 

311 def metadata(self): 

312 items = {} 

313 keyrings = (self.metadata_required_keys, self.metadata_optional_keys,) 

314 for keyring in keyrings: 

315 for key in keyring: 

316 # TODO On refactoring of the metadata properties, 

317 # required fields will raise MissingMetadataError, 

318 # but for now do that here. 

319 value = getattr(self, key) 

320 if key in self.metadata_required_keys and value is None: 

321 raise MissingMetadataError( 

322 "A value for '{}' could not be found.".format(key)) 

323 elif value is None: 

324 continue 

325 items[key] = value 

326 return items

329class Package(Sequence): 

330 """EPUB3 package""" 

331 

332 def __init__(self, name, items, metadata=None): 

333 self.name = name 

334 self.metadata = metadata or {} 

335 self._items = items 

336 navigation_items = [i for i in self._items if i.is_navigation] 

337 if len(navigation_items) == 0: 

338 raise MissingNavigationError("Navigation item not found") 

339 elif len(navigation_items) > 1: 

340 raise AdditionalNavigationError( 

341 "Only one navigation item can exist " 

342 "per package. The given value is a second " 

343 "navigation item.") 

344 else: 

345 index = self._items.index(navigation_items[0]) 

346 self._navigation_item_index = index 

347 

348 @classmethod 

349 def from_file(cls, file): 

350 """Create the object from a *file* or *file-like object*.""" 

351 opf_xml = etree.parse(file) 

352 # Check if ``file`` is file-like. 

353 if hasattr(file, 'read'): 

354 name = os.path.basename(file.name) 

355 root = os.path.abspath(os.path.dirname(file.name)) 

356 else: # ...a filepath 

357 name = os.path.basename(file) 

358 root = os.path.abspath(os.path.dirname(file)) 

359 parser = OPFParser(opf_xml) 

360 

361 # Roll through the item entries 

362 manifest = opf_xml.xpath('/opf:package/opf:manifest/opf:item', 

363 namespaces=EPUB_OPF_NAMESPACES) 

364 pkg_items = [] 

365 for item in manifest: 

366 absolute_filepath = os.path.join(root, item.get('href')) 

367 properties = item.get('properties', '').split() 

368 is_navigation = 'nav' in properties 

369 media_type = item.get('media-type') 

370 pkg_items.append(Item.from_file(absolute_filepath, 

371 media_type=media_type, 

372 is_navigation=is_navigation, 

373 properties=properties)) 

374 # Ignore spine ordering, because it is not important 

375 # for our use cases. 

376 return cls(name, pkg_items, parser.metadata) 

377 

378 @staticmethod 

379 def to_file(package, directory): 

380 """Write the package to the given ``directory``. 

381 Returns the OPF filename. 

382 """ 

383 opf_filepath = os.path.join(directory, package.name) 

384 

385 # Create the directory structure 

386 for name in ('contents', 'resources',): 

387 path = os.path.join(directory, name) 

388 if not os.path.exists(path): 

389 os.mkdir(path) 

390 

391 # Write the items to the filesystem 

392 locations = {} # Used when rendering 

393 for item in package: 

394 if item.media_type == 'application/xhtml+xml': 

395 base = os.path.join(directory, 'contents') 

396 else: 

397 base = os.path.join(directory, 'resources') 

398 filename = item.name 

399 filepath = os.path.join(base, filename) 

400 locations[item] = os.path.relpath(filepath, directory) 

401 with open(filepath, 'wb') as item_file: 

402 item_file.write(item.data.read()) 

403 

404 # Write the OPF 

405 template = jinja2.Template(OPF_TEMPLATE, 

406 trim_blocks=True, lstrip_blocks=True) 

407 with open(opf_filepath, 'wb') as opf_file: 

408 opf = template.render(package=package, locations=locations) 

409 if not isinstance(opf, bytes): 

410 opf = opf.encode('utf-8') 

411 opf_file.write(opf) 

412 

413 return opf_filepath
