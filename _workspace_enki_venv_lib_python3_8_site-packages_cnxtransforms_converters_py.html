148def html_to_cnxml(html, xml_parser=DEFAULT_XMLPARSER): 

149 """Transform html content to cnxml.""" 

150 if not isinstance(html, (etree._ElementTree, etree._Element,)): 

151 html = etree.parse(BytesIO(ensure_bytes(html)), xml_parser).getroot() 

152 elif isinstance(html, etree._ElementTree): 

153 html = html.getroot() 

154 

155 # Do the namespace dance... 

156 nsmap = html.nsmap.copy() 

157 nsmap['h'] = nsmap.pop(None) 

158 

159 # The transform only works when the root tag is 'body'. 

160 if html.xpath('/h:html', namespaces=nsmap): 

161 html = html.xpath('/h:html/h:body', namespaces=nsmap)[0] 

162 

163 # Transform the content. 

164 cnxml = HTML_TO_CNXML_XSL(html) 

165 return etree.tostring(cnxml)
