186def parse_legacy_reference(ref): 

187 """Parse legacy reference into parts. Returns a tuple. 

188 

189 Parse the legacy reference to a reference type and type specific value. 

190 A module-reference value contains the id, version and fragment. 

191 A resource-reference value contains resource filename. 

192 """ 

193 match = LEGACY_PATH_REFERENCE_REGEX.match(ref) 

194 try: 

195 # Dictionary keyed by named groups, None values for no match 

196 matches = match.groupdict() 

197 except AttributeError: # None type 

198 raise ValueError("Unable to parse reference with value '{}'" 

199 .format(ref)) 

200 

201 version = matches['version'] 

202 if version == 'latest': 

203 version = None 

204 collection_version = matches['collection_version'] 

205 if collection_version == 'latest': 

206 collection_version = None 

207 

208 # We've got a match, but what kind of thing is it. 

209 if matches['legacy']: 

210 # Don't transform legacy urls if hostname is legacy.cnx.org 

211 type = None 

212 value = () 

213 elif matches['resource']: 

214 type = RESOURCE_REFERENCE 

215 value = (matches['resource'].strip(), matches['module'], version) 

216 elif matches['module']: 

217 type = MODULE_REFERENCE 

218 value = (matches['module'], version, matches['collection'], 

219 collection_version, matches['fragment']) 

220 else: 

221 type = None 

222 value = () 

223 return type, value

227 """Parse HTML cross reference. 

228 

229 Parse the html reference to a reference type and type specific value. 

230 A document-reference value contains the ident-hash. 

231 A binder-reference value contains the ident-hash and possibly a document 

232 ident-hash. 

233 A resource-reference value resource filename. 

234 """ 

235 match = PATH_REFERENCE_REGEX.match(ref) 

236 try: 

237 # Dictionary keyed by named groups, None values for no match 

238 matches = match.groupdict() 

239 except AttributeError: # None type 

240 raise ValueError("Unable to parse reference with value '{}'" 

241 .format(ref)) 

242 

243 try: 

244 version = matches['version'] 

245 except KeyError: 

246 version = None 

247 

248 # We've got a match, but what kind of thing is it. 

249 if matches['legacy']: 

250 # Don't transform legacy urls if hostname is legacy.cnx.org 

251 type = None 

252 value = () 

253 elif matches['resource']: 

254 type = RESOURCE_REFERENCE 

255 value = (matches['resource'], matches['fragment'],) 

256 elif matches['bound_document']: 

257 type = BINDER_REFERENCE

259 matches['fragment'],)

261 # Note, this could also be a binder. 

262 type = DOCUMENT_REFERENCE 

263 value = (matches['document'], version, matches['fragment'],) 

264 else: 

265 type = None 

266 value = () 

267 return type, value

391 def get_page_ident_hash(self, page_uuid, page_version, 

392 book_uuid, book_version, latest=None): 

393 """Return the uuid of the page and full ident_hash of the page. 

394 

395 which may or may not include the book uuid depending on whether 

396 the page is within the book. 

397 """ 

398 from cnxepub import flatten_tree_to_ident_hashes # XXX 

399 plan = self.plpy.prepare( 

400 'SELECT tree_to_json($1, $2, FALSE)::json', ('text', 'text')) 

401 tree = self.plpy.execute( 

402 plan, (book_uuid, book_version))[0]['tree_to_json'] 

403 if isinstance(tree, basestring):

414 latest and book_uuid or book_ident_hash, 

415 page_ident_hash) 

416 # The page isn't in the given book, so only return the page. 

417 return page_uuid, page_ident_hash

419 def fix_media_references(self): 

420 """Fix references to interal resources.""" 

421 # Catch the invalid, unparsable, etc. references. 

422 bad_references = [] 

423 

424 media_xpath = { 

425 '//html:img': ('src', 'longdesc',), 

426 '//html:audio': ('src',), 

427 '//html:video': ('src',), 

428 '//html:object': ('data',), 

429 '//html:object/html:embed': ('src',),

454 .format(info['hash'], filename))

590 def fix_link_references(self): 

591 """Fix references to internal documents and resources.""" 

592 # Catch the invalid, unparsable, etc. references. 

593 bad_references = [] 

594 

595 # Note, all c:link will have an @url. We purposely dumb down the xslt 

596 # in order to make the scan here easy. Plus the xslt could never 

597 # fully match and disassemble the url into the various attributes on 

598 # a link tag. 

599 for link in self.apply_xpath('//c:link'): 

600 ref = link.get('url') 

601 

602 if not ref or self._should_ignore_reference(ref): 

603 continue 

604 

605 try: 

606 ref_type, payload = parse_html_reference(ref) 

607 except ValueError: 

608 exc = InvalidReference(self.document_ident, ref) 

609 bad_references.append(exc) 

610 continue 

611 

612 # Delete the tentative attribute 

613 link.attrib.pop('url') 

614 

615 # TODO handle #{id} refs in the url_frag, which could also contain 

616 # path elements, so further parsing is necessary. 

617 

618 if ref_type == DOCUMENT_REFERENCE: 

619 id, version, url_frag = payload 

620 mid, version = self.get_mid_n_version(id, version) 

621 if mid is None: 

622 bad_references.append( 

623 ReferenceNotFound("Unable to find a reference to " 

624 "'{}' at version '{}'." 

625 .format(id, version), 

626 self.document_ident, ref)) 

627 else: 

628 # Assign the document specific attributes. 

629 link.attrib['document'] = mid 

630 if version is not None: 

631 link.attrib['version'] = version 

632 elif ref_type == BINDER_REFERENCE: 

633 id, version, bound_document, url_frag = payload 

634 mid, version = self.get_mid_n_version(id, version) 

635 if mid is None: 

636 id, version, id_type = split_ident_hash(bound_document) 

637 cid, cversion = self.get_mid_n_version(id, version) 

638 if cid is None: 

639 # Binder ref doesn't exist, but Document does. 

640 # Assign the document specific attributes. 

641 link.attrib['document'] = mid 

642 if version is not None: 

643 link.attrib['version'] = version 

644 else: 

645 # Process the ref as an external url. 

646 url = "http://legacy.cnx.org/content/" \ 

647 "{}/{}/?collection={}".format(mid, version, cid) 

648 if cversion is not None: 

649 url = "{}/{}".format(url, cversion) 

650 link.attrib['url'] = url 

651 else: 

652 bad_references.append( 

653 ReferenceNotFound("Unable to find a reference to " 

654 "'{}' at version '{}'." 

655 .format(id, version), 

656 self.document_ident, ref)) 

657 elif ref_type == RESOURCE_REFERENCE: 

658 # TODO If the filename is in the url_frag, use it 

659 # as the filename instead of looking it up. 

660 try: 

661 sha1_hash, url_frag = payload 

662 filename = self.get_resource_filename(sha1_hash) 

663 except ReferenceNotFound as exc: 

664 bad_references.append(exc) 

665 else: 

666 link.attrib['resource'] = filename 

667 else: 

668 exc = InvalidReference(self.document_ident, ref) 

669 bad_references.append(exc) 

670 # Preserve the content value... 

671 link.attrib['url'] = ref 

672

676 """Fix references to interal resources.""" 

677 # Catch the invalid, unparsable, etc. references. 

678 bad_references = [] 

679 

680 media_xpath = {

682 '//c:image': ('src', 'thumbnail',), 

683 '//c:audio': ('src',),

685 '//c:java-applet': ('src',), 

686 '//c:flash': ('src',), 

687 '//c:download': ('src',), 

688 '//c:labview': ('src',), 

689 } 

690 

691 for xpath, attrs in media_xpath.items(): 

692 for elem in self.apply_xpath(xpath): 

693 for attr in attrs: 

694 ref = elem.get(attr) 

695 if not ref or self._should_ignore_reference(ref): 

696 continue 

697 try: 

698 ref_type, payload = parse_html_reference(ref) 

699 sha1_hash, url_frag = payload 

700 except ValueError:

702 bad_references.append(exc) 

703 continue 

704 try: 

705 filename = self.get_resource_filename(sha1_hash) 

706 except ReferenceNotFound as exc: 

707 bad_references.append(exc) 

708 else: 

709 elem.set(attr, filename) 

710 return bad_references
