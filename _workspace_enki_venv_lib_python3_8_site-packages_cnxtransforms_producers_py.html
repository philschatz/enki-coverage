74def produce_html_for_abstract(plpy, document_ident): 

75 """Produce html for the abstract by the given ``document_ident``.""" 

76 plan = plpy.prepare("SELECT abstractid, abstract " 

77 "FROM modules NATURAL LEFT JOIN abstracts " 

78 "WHERE module_ident = $1;", ('integer',)) 

79 try: 

80 result = plpy.execute(plan, (document_ident,), 1)[0] 

81 abstractid, abstract = result['abstractid'], result['abstract'] 

82 except (IndexError, KeyError): 

83 # This means the document doesn't exist. 

84 raise ValueError("No document at ident: {}".format(document_ident)) 

85 if abstractid is None: 

86 raise MissingAbstract(document_ident) 

87 

88 warning_messages = None 

89 # Transform the abstract. 

90 if abstract: 

91 html, warning_messages = transform_abstract_to_html( 

92 abstract, document_ident, plpy) 

93 else: 

94 html = None 

95 

96 # Update the abstract. 

97 if html: 

98 html = html.decode('utf-8') 

99 plan = plpy.prepare("UPDATE abstracts SET (html) = ($1) " 

100 "WHERE abstractid = $2;", ('text', 'integer')) 

101 plpy.execute(plan, (html, abstractid,)) 

102 return warning_messages

137def transform_abstract_to_html(abstract, document_ident, plpy): 

138 """Convert abstract to html.""" 

139 warning_messages = None 

140 abstract_html = cnxml_abstract_to_html(abstract)

142 # Then fix up content references in the abstract.

153def transform_abstract_to_cnxml(abstract, document_ident, plpy): 

154 """Transform an html abstract to cnxml.""" 

155 warning_messages = None 

156 cnxml = None 

157 

158 if abstract: 

159 # Mark the wrapping 'div' as an identifiable data-type. 

160 html = etree.fromstring(abstract) 

161 container = html.xpath('/*')[0] 

162 container.attrib['data-type'] = 'abstract-wrapper' 

163 # Transform the abstract. 

164 cnxml = html_to_cnxml(html) 

165 

166 # Then fix up content references in the abstract. 

167 if cnxml: 

168 cnxml, bad_refs = resolve_html_urls(BytesIO(cnxml), 

169 plpy, document_ident) 

170 if bad_refs: 

171 warning_messages = 'Invalid References (Abstract): {}' \ 

172 .format('; '.join(bad_refs)) 

173 

174 # Strip the <wrapper> wrapper tag off the content. 

175 # The wrapper is used so that bare text can be in the abstract 

176 # and still be valid xml, which is required for the reference resolver. 

177 start = cnxml.find(b'>') + 1 

178 end = len(cnxml) - len('</wrapper>') 

179 cnxml = cnxml[start:end] 

180 

181 return cnxml, warning_messages

231def produce_transformed_file(plpy, ident, transform_type, 

232 source_filename, destination_filenames, 

233 overwrite=False): 

234 """Produce a file from a given module source file with a specifc transform. 

235 

236 Produce an ``destination_filename`` file for the module at ``ident`` 

237 using the ``source_filename``. 

238 

239 Raises exceptions when the transform cannot be completed. 

240 

241 Returns a message containing warnings and other information that 

242 does not effect the content, but may affect the user experience 

243 of it. 

244 """ 

245 transform_info = TRANSFORM_TYPES[transform_type] 

246 transformer, reference_resolver, media_type = transform_info 

247 plan = plpy.prepare("SELECT convert_from(file, 'utf-8') " 

248 "FROM module_files " 

249 " NATURAL LEFT JOIN files " 

250 "WHERE module_ident = $1 " 

251 " AND filename = $2;", ('integer', 'text')) 

252 result = plpy.execute(plan, (ident, source_filename,)) 

253 try: 

254 content = result[0]['convert_from'][:] # returns: (<bufferish ...>,) 

255 except (IndexError, KeyError): # None returned 

256 raise MissingDocumentOrSource(ident, source_filename) 

257 

258 # Remove destination if overwrite is True and if it exists 

259 plan = plpy.prepare('SELECT fileid FROM module_files ' 

260 'WHERE module_ident = $1 ' 

261 ' AND filename = ANY($2)', 

262 ('integer', 'text[]')) 

263 file_id_rows = plpy.execute(plan, (ident, list(destination_filenames),)) 

264 for row in file_id_rows: 

265 file_id = row['fileid'] 

266 if overwrite: 

267 delete_module_files = plpy.prepare( 

268 'DELETE FROM module_files WHERE fileid = $1', ('integer',)) 

269 plpy.execute(delete_module_files, (file_id,)) 

270 delete_files = plpy.prepare( 

271 'DELETE FROM files WHERE fileid = $1', ('integer',)) 

272 plpy.execute(delete_files, (file_id,)) 

273 else: 

274 raise IndexFileExistsError(ident, destination_filenames) 

275 

276 new_content = ensure_bytes(transformer(content)) 

277 

278 # Fix up content references to cnx-archive specific urls. 

279 new_content, bad_refs = reference_resolver(BytesIO(new_content), 

280 plpy, ident) 

281 

282 warning_messages = None 

283 if bad_refs: 

284 warning_messages = 'Invalid References: {}' \ 

285 .format('; '.join(bad_refs)) 

286 

287 # Find existing file in database before attempting to insert a new one. 

288 sha1 = hashlib.new('sha1', new_content).hexdigest() 

289 plan = plpy.prepare("SELECT fileid FROM files WHERE sha1 = $1;", 

290 ('text',)) 

291 try: 

292 destination_file_id = plpy.execute(plan, (sha1,))[0]['fileid'] 

293 except IndexError: 

294 # Insert the cnxml into the database. 

295 payload = (new_content, media_type,) 

296 plan = plpy.prepare("INSERT INTO files (file, media_type) " 

297 "VALUES ($1, $2) " 

298 "RETURNING fileid;", ('bytea', 'text')) 

299 destination_file_id = plpy.execute(plan, payload)[0]['fileid'] 

300 for filename in destination_filenames: 

301 insert_module_files = plpy.prepare( 

302 "INSERT INTO module_files " 

303 " (module_ident, fileid, filename) " 

304 " VALUES ($1, $2, $3);", 

305 ('integer', 'integer', 'text')) 

306 plpy.execute(insert_module_files, 

307 (ident, destination_file_id, filename)) 

308 return warning_messages
