20def parse_book_tree(bookdir): 

21 """Converts filesystem booktree back to a struct""" 

22 struct = [] 

23 ex = [lambda filepath: '.sha1sum' in filepath.name] 

24 ex.extend([lambda filepath: 'metadata.json' in filepath.name]) 

25 for dirname, subdirs, filenames in os.walk(str(bookdir)): 

26 if 'collection.xml' in filenames: 

27 path = Path(dirname) 

28 sha1s = get_sha1s_dict(path) 

29 struct.append((parse_collection(path, excludes=ex), sha1s)) 

30 elif 'index.cnxml' in filenames: 

31 path = Path(dirname) 

32 sha1s = get_sha1s_dict(path) 

33 struct.append((parse_module(path, excludes=ex), sha1s)) 

34 

35 return struct

48def filter_what_changed(contents): 

49 changed = [] 

50 

51 collection, coll_sha1s_dict = contents.pop(0) 

52 

53 for model, sha1s_dict in contents: 

54 new_mod_resources = [] 

55 for resource in model.resources: 

56 cached_sha1 = sha1s_dict.get(resource.filename.strip()) 

57 

58 if cached_sha1 is None or resource.sha1 != cached_sha1: 

59 new_mod_resources.append(resource) 

60 

61 # if the model changed or any of its resources 

62 mod_cached_sha1 = sha1s_dict.get(model.file.name.strip()) 

63 mod_actual_sha1 = calculate_sha1(model.file) 

64 if mod_actual_sha1 != mod_cached_sha1 or len(new_mod_resources) > 0: 

65 new_model = model._replace(resources=tuple(new_mod_resources)) 

66 changed.append(new_model) 

67 

68 """Now check the Collection and the Collection's resources""" 

69 new_col_resources = [] 

70 for resource in collection.resources: 

71 cached_sha1 = coll_sha1s_dict.get(resource.filename.strip()) 

72 

73 if cached_sha1 is None or resource.sha1 != cached_sha1: 

74 new_col_resources.append(resource) 

75 

76 coll_changed = False 

77 new_coll = collection._replace(resources=tuple(new_col_resources)) 

78 if len(new_col_resources) > 0: 

79 changed.insert(0, new_coll) # because `_publish` will expect this 

80 coll_changed = True 

81 

82 # If any modules changed, assume collection changed 

83 cached_coll_sha1 = coll_sha1s_dict.get('collection.xml') 

84 if len(changed) > 0 or coll_sha1s_dict.get('collection.xml') is None or \ 

85 cached_coll_sha1 != calculate_sha1(collection.file): 

86 if not coll_changed: # coll already in changed. 

87 changed.insert(0, new_coll) 

88 return changed 

89 else: # No changes at all 

90 return []
