89def init_repository(path, bare=False, 

90 flags=GIT_REPOSITORY_INIT_MKPATH, 

91 mode=0, 

92 workdir_path=None, 

93 description=None, 

94 template_path=None, 

95 initial_head=None, 

96 origin_url=None): 

97 """ 

98 Creates a new Git repository in the given *path*. 

99 

100 If *bare* is True the repository will be bare, i.e. it will not have a 

101 working copy. 

102 

103 The *flags* may be a combination of: 

104 

105 - GIT_REPOSITORY_INIT_BARE (overriden by the *bare* parameter) 

106 - GIT_REPOSITORY_INIT_NO_REINIT 

107 - GIT_REPOSITORY_INIT_NO_DOTGIT_DIR 

108 - GIT_REPOSITORY_INIT_MKDIR 

109 - GIT_REPOSITORY_INIT_MKPATH (set by default) 

110 - GIT_REPOSITORY_INIT_EXTERNAL_TEMPLATE 

111 

112 The *mode* parameter may be any of GIT_REPOSITORY_SHARED_UMASK (default), 

113 GIT_REPOSITORY_SHARED_GROUP or GIT_REPOSITORY_INIT_SHARED_ALL, or a custom 

114 value. 

115 

116 The *workdir_path*, *description*, *template_path*, *initial_head* and 

117 *origin_url* are all strings. 

118 

119 See libgit2's documentation on git_repository_init_ext for further details. 

120 """ 

121 # Pre-process input parameters 

122 if path is None: 

123 raise TypeError('Expected string type for path, found None.') 

124 

125 if bare: 

126 flags |= GIT_REPOSITORY_INIT_BARE 

127 

128 # Options 

129 options = ffi.new('git_repository_init_options *') 

130 C.git_repository_init_init_options(options, 

131 GIT_REPOSITORY_INIT_OPTIONS_VERSION) 

132 options.flags = flags 

133 options.mode = mode 

134 

135 if workdir_path: 

136 workdir_path_ref = ffi.new('char []', to_bytes(workdir_path)) 

137 options.workdir_path = workdir_path_ref 

138 

139 if description: 

140 description_ref = ffi.new('char []', to_bytes(description)) 

141 options.description = description_ref 

142 

143 if template_path: 

144 template_path_ref = ffi.new('char []', to_bytes(template_path)) 

145 options.template_path = template_path_ref 

146 

147 if initial_head: 

148 initial_head_ref = ffi.new('char []', to_bytes(initial_head)) 

149 options.initial_head = initial_head_ref 

150 

151 if origin_url: 

152 origin_url_ref = ffi.new('char []', to_bytes(origin_url)) 

153 options.origin_url = origin_url_ref 

154 

155 # Call 

156 crepository = ffi.new('git_repository **') 

157 err = C.git_repository_init_ext(crepository, to_bytes(path), options) 

158 check_error(err) 

159 

160 # Ok 

161 return Repository(to_str(path))

164def clone_repository( 

165 url, path, bare=False, repository=None, remote=None, 

166 checkout_branch=None, callbacks=None): 

167 """ 

168 Clones a new Git repository from *url* in the given *path*. 

169 

170 Returns: a Repository class pointing to the newly cloned repository. 

171 

172 Parameters: 

173 

174 url : str 

175 URL of the repository to clone. 

176 path : str 

177 Local path to clone into. 

178 bare : bool 

179 Whether the local repository should be bare. 

180 remote : callable 

181 Callback for the remote to use. 

182 

183 The remote callback has `(Repository, name, url) -> Remote` as a 

184 signature. The Remote it returns will be used instead of the default 

185 one. 

186 repository : callable 

187 Callback for the repository to use. 

188 

189 The repository callback has `(path, bare) -> Repository` as a 

190 signature. The Repository it returns will be used instead of creating a 

191 new one. 

192 checkout_branch : str 

193 Branch to checkout after the clone. The default is to use the remote's 

194 default branch. 

195 callbacks : RemoteCallbacks 

196 Object which implements the callbacks as methods. 

197 

198 The callbacks should be an object which inherits from 

199 `pyclass:RemoteCallbacks`. 

200 """ 

201 

202 if callbacks is None: 

203 callbacks = RemoteCallbacks() 

204 

205 # Add repository and remote to the payload 

206 payload = callbacks 

207 payload.repository = repository 

208 payload.remote = remote 

209 

210 with git_clone_options(payload): 

211 opts = payload.clone_options 

212 opts.bare = bare 

213 

214 if checkout_branch: 

215 checkout_branch_ref = ffi.new('char []', to_bytes(checkout_branch)) 

216 opts.checkout_branch = checkout_branch_ref 

217 

218 with git_fetch_options(payload, opts=opts.fetch_opts): 

219 crepo = ffi.new('git_repository **') 

220 err = C.git_clone(crepo, to_bytes(url), to_bytes(path), opts) 

221 payload.check_error(err) 

222 

223 # Ok 

224 return Repository._from_c(crepo[0], owned=True)
