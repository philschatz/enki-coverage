22def get_checksums(filename): 

23 """ generate SHA1 and S3 MD5 etag checksums from file """ 

24 sha1 = hashlib.sha1() 

25 md5s = [] 

26 try: 

27 with open(filename, 'rb') as f: 

28 while True: 

29 data = f.read(BUF_SIZE) 

30 if not data: 

31 break 

32 sha1.update(data) 

33 md5s.append(hashlib.md5(data)) 

34 # chunked calculation for AWS S3 MD5 etag 

35 # https://stackoverflow.com/a/43819225/756056 

36 # 

37 # AWS needs the MD5 quoted inside the string json value. 

38 # Despite looking like a mistake, this is correct behavior. 

39 if len(md5s) < 1: 

40 s3_md5 = '"{}"'.format(hashlib.md5().hexdigest()) # pragma: no cover 

41 elif len(md5s) == 1: 

42 s3_md5 = '"{}"'.format(md5s[0].hexdigest()) 

43 else: # pragma: no cover 

44 digests = b''.join(m.digest() for m in md5s) 

45 digests_md5 = hashlib.md5(digests) 

46 s3_md5 = '"{}-{}"'.format(digests_md5.hexdigest(), len(md5s)) 

47 return sha1.hexdigest(), s3_md5 

48 except IOError: # file does not exist 

49 return None, None

112def ensure_isoformat(timestamp): 

113 """Given a timestsamp string either validate it is already ISO8601 and 

114 return or attempt to convert it. 

115 """ 

116 try: 

117 # Using dateutil.parser here instead of datetime.fromisoformat as the 

118 # former seems to be more robust and avoids some false negatives seen 

119 # with the latter (e.g. with "2021-03-22T14:14:33.17588-05:00") 

120 parser.isoparse(timestamp) 

121 return timestamp 

122 except ValueError: 

123 # The provided timestamp needs to be converted 

124 pass 

125 

126 # Try parsing timezone separately to catch cases like 'GMT-5' and 

127 # 'US/Central' 

128 # 

129 # Note: We're attempting this before dateutil.parser.parse because if we 

130 # don't cases like GMT-5 will end up as offset +5:00 

131 # (see https://github.com/dateutil/dateutil/issues/70) 

132 try: 

133 timestamp_notz, timestamp_tz = timestamp.rsplit(' ', 1) 

134 parsed_tz = tz.gettz(timestamp_tz) 

135 # The parsed timezone may be None (e.g. for '-0500') 

136 if parsed_tz: 

137 return parser.parse(timestamp_notz).replace(tzinfo=parsed_tz).isoformat() 

138 except ValueError: 

139 pass 

140 

141 # Final attempt with just dateutil parser 

142 try: 

143 return parser.parse(timestamp).isoformat() 

144 except ValueError: 

145 pass 

146 

147 raise Exception(f"Could not convert non ISO8601 timestamp: {timestamp}")
